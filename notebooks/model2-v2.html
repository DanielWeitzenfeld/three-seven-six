#hyper parameters
#mu_skill = pymc.Normal('mu_skill', likely_intercept, .0001, value=likely_intercept)
mu_skill = pymc.Normal('mu_skill', 0, .0001, value=likely_intercept)
tau_skill = pymc.Uniform('tau_skill', lower=0, upper=100)

# top-level parameters
injury = pymc.Normal('injury', mu=0, tau=.0001, value=0)
corner = pymc.Normal('corner', mu=0, tau=.0001, value=0)

# player-specific parameters
skill = pymc.Normal('skill', mu=mu_skill, tau=tau_skill, size=num_shooters)

def logistic(x):
    return 1 / (1 + math.exp(-x))

@pymc.deterministic
def arc_thetas( player_season_crosswalk=player_season_crosswalk,
               observed_season_interruptions=observed_season_interruptions,
          skill=skill, 
          injury=injury):
    thetas = []
    for season, player in enumerate(player_season_crosswalk):
        innate_theta = skill[player]
        interrupted_season_dummy = observed_season_interruptions[season]
        arc_theta = logistic(innate_theta + injury*interrupted_season_dummy)
        thetas.append(arc_theta)
    return thetas

@pymc.deterministic
def corner_thetas( player_season_crosswalk=player_season_crosswalk,
               observed_season_interruptions=observed_season_interruptions,
          skill=skill, 
          injury=injury,
          corner=corner):
    thetas = []
    for season, player in enumerate(player_season_crosswalk):
        innate_theta = skill[player]
        interrupted_season_dummy = observed_season_interruptions[season]
        corner_theta = logistic(innate_theta + injury*interrupted_season_dummy + corner)
        thetas.append( corner_theta)
    return thetas
    
    
arc_makes = pymc.Binomial('arc_makes', value=observed_arc_makes, n=observed_arc_attempts, p=arc_thetas, observed=True)
corner_makes = pymc.Binomial('corner_makes', value=observed_corner_makes, n=observed_corner_attempts, p=corner_thetas, observed=True) 

mcmc = pymc.MCMC([arc_makes, corner_makes, arc_thetas, corner_thetas, skill, injury, corner,

map_ = pymc.MAP( mcmc )

def fetch_player_name(player_index):
    return df.player[df.player_index == player_index].values[0]

def plot_player_trace(player_index):
    player_name = fetch_player_name(player_index)
    n, bins, patches = plt.hist(player_traces[:, player_index])
    plt.title(player_name)

def player_trace_frame(player_index):
    skill_trace = player_traces[:, player_index]
    corner_trace = corner.trace.gettrace()  
    df_traces = pd.DataFrame({'corner': corner_trace, 'skill': skill_trace})
    df_traces['predicted_arc_percentage'] = df_traces.apply(lambda x: logistic( x['skill']), axis=1)
    df_traces['predicted_corner_percentage'] = df_traces.apply(lambda x: logistic( x['skill'] + x['corner']), axis=1)
    return df_traces


XLIM = (.2, .55)

def visualize_player(player_index, last_n_trace_samples=1000, samples_per_theta = 10, jitter=JITTER):
    player_name = fetch_player_name(player_index)
    df_traces = player_trace_frame(player_index)  
    median_arc_attempts = df.arc_3pfga[df.player_index==player_index].median()
    median_corner_attempts = df.corner_3pfga[df.player_index==player_index].median()
    #take samples_per_theta samples from each sampled theta to represent measurement noise
    xs=[]
    ys=[]
    for i, row in df_traces[-last_n_trace_samples:].iterrows():
        for j in range(samples_per_theta):
            xs += list(np.random.binomial(median_arc_attempts, row.predicted_arc_percentage, samples_per_theta)/median_arc_attempts)
            ys += list(np.random.binomial(median_corner_attempts, row.predicted_corner_percentage, samples_per_theta)/median_corner_attempts)
    if JITTER:
        xs = np.array(xs)
        xs = xs + np.random.uniform(0, JITTER_AMOUNT, size=len(xs))
        ys = np.array(ys)
        ys = ys + np.random.uniform(0, JITTER_AMOUNT, size=len(ys))
    fig, axs = plt.subplots()
    ax = plt.scatter(xs, ys, alpha=ALPHA, c=RED, s=5)
    season_data = df.ix[(df.player_index==player_index) & 
                        (df.interrupted_season == 0)]
    plt.scatter(season_data.fgpc_3parc, 
                season_data.fpgc_3pcorner , 
                s=(SEASON_BUBBLE_SIZE*season_data.attempts_as_percent_of_max**2) * np.pi, c=BLUE)
    for i, s in season_data.iterrows():
        plt.annotate(xy=(s['fgpc_3parc'], s['fpgc_3pcorner']), s=s['year'], 
                     xytext=OFFSET, textcoords = 'offset points', color=BLUE, weight=WEIGHT, size=SIZE)
    plt.title('%s: %s' % (player_name, TITLE_ALL_YEARS))
    plt.ylabel('Corner Three Point Percentage')
    plt.xlabel('Arc Three Point Percentage')
    axs.set_ylim(YLIM)
    axs.set_xlim(XLIM)
    return ax
